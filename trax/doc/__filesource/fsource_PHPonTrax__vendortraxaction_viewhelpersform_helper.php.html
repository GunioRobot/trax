  148.     function to_check_box_tag($options = array(), $checked_value = "1", $unchecked_value = "0") {
    149. 

  149.         $options["type"] = "checkbox";
    150. 

  150.         $options["value"] = $checked_value;
    151. 

  151.         switch(gettype($this->value())) {
    152. 

  152.             case 'boolean':
    153. 

  153.                 $checked = $this->value();
    154. 

  154.                 break;
    155. 

  155.             case 'NULL':
    156. 

  156.                 $checked = false;
    157. 

  157.                 break;
    158. 

  158.             case 'integer':

  159.                 $checked = ($this->value() != 0);
    160. 

  160.                 break;
    161. 

  161.             case 'string':
    162. 

  162.                 $checked = ($this->value() == $checked_value);
    163. 

  163.                 break;
    164. 

  164.             default:
    165. 

  165.                 $checked = ($this->value() != 0);
    166. 

  166.         }
    167. 

  167.  
    168. 

  168.         if ($checked || $options["checked"] == "checked") {
    169. 

  169.             $options["checked"] = "checked";
    170. 

  170.         } else {
    171. 

  171.             unset($options["checked"]);
    172. 

  172.         }
    173. 

  173.  
    174. 

  174.         $options = $this->add_default_name_and_id($options);
    175. 

  175.         return $this->error_wrapping($this->tag("input", $options) . $this->tag("input", array("name" => $options["name"], "type" => "hidden", "value" => $unchecked_value)),$this->object()->errors[$this->attribute_name]);

  254. /**
    255. 

  255.  * Returns a checkbox tag tailored for accessing a specified attribute (identified by $field) on an object
    256. 

  256.  * assigned to the template (identified by $object). It's intended that $field returns an integer and if that
    257. 

  257.  * integer is above zero, then the checkbox is checked. Additional $options on the input tag can be passed as an
    258. 

  258.  * array with $options. The $checked_value defaults to 1 while the default $unchecked_value
    259. 

  259.  * is set to 0 which is convenient for boolean values. Usually unchecked checkboxes don't post anything.
    260. 

  260.  * We work around this problem by adding a hidden value with the same name as the checkbox.
    261. 

  261. #
    262. 

  262.  * Example: Imagine that $post->validated is 1:
    263. 

  263.  *   check_box("post", "validated");
    264. 

  264.  * Result:
    265. 

  265.  *   <input type="checkbox" id="post_validate" name="post[validated] value="1" checked="checked" />
    266. 

  266.  *   <input name="post[validated]" type="hidden" value="0" />
    267. 

  267. #
    268. 

  268.  * Example: Imagine that $puppy->gooddog is no:
    269. 

  269.  *   check_box("puppy", "gooddog", array(), "yes", "no");
    270. 

  270.  * Result:
    271. 

  271.  *     <input type="checkbox" id="puppy_gooddog" name="puppy[gooddog] value="yes" />
    272. 

  272.  *     <input name="puppy[gooddog]" type="hidden" value="no" />
    273. 

  273. */
    274.
