  37.     /**
    38. 

  38.      * Creates a link tag of the given +name+ using an URL created by the set of +options+.
    39. 

  39.      * It's also possible to pass a string instead of an options hash to
    40. 

  40.      * get a link tag that just points without consideration. If null is passed as a name, the link itself will become the name.
    41. 

  41.      * The $html_options have a special feature for creating javascript confirm alerts where if you pass ":confirm" => 'Are you sure?',
    42. 

  42.      * the link will be guarded with a JS popup asking that question. If the user accepts, the link is processed, otherwise not.
    43. 

  43.      *
    44. 

  44.      * Example:
    45. 

  45.      *   link_to("Delete this page", array(":action" => "delete", ":id" => $page->id), array(":confirm" => "Are you sure?"))
    46. 

  46.      */
    47. 

  47.     function link_to($name, $options = array(), $html_options = array()) {
    48. 

  48.         $html_options = $this->convert_confirm_option_to_javascript($html_options);
    49. 

  49.         if(is_string($options)) {
    50. 

  50.             $href = array("href" => $options);
    51. 

  51.             if(count($html_options) > 0) {
    52. 

  52.                 $html_options = array_merge($html_options, $href);
    53. 

  53.             } else {
    54. 

  54.                 $html_options = $href;
    55. 

  55.             }
    56. 

  56.             if(!$name) {
    57. 

  57.                 $name = $options;
    58. 

  58.             }
    59. 

  59.             $html = $this->content_tag("a", $name, $html_options);
    60. 

  60.         } else {
    61. 

  61.             $url = $this->url_for($options);
    62. 

  62.             if(!$name) {
    63. 

  63.                 $name = $url;
    64. 

  64.             }
    65. 

  65.             $href = array("href" => $url);
    66. 

  66.             if(count($html_options) > 0) {
    67. 

  67.                 $html_options = array_merge($html_options, $href);
    68. 

  68.             } else {
    69. 

  69.                 $html_options = $href;
    70. 

  70.             }
    71. 

  71.             $html = $this->content_tag("a", $name, $html_options);
    72. 

  72.         }
    73. 

  73.         return $html;
    74. 

  74.     }

  102.     function button_to($name, $options = array(), $html_options = null) {
    103. 

  103.         $html_options = (!is_null($html_options) ? $html_options : array());
    104. 

  104.         $this->convert_boolean_attributes($html_options, array('disabled'));
    105. 

  105.         $this->convert_confirm_option_to_javascript($html_options);

  106.         if (is_string($options)) {
    107. 

  107.             $url = $options;

  108.             $name = (!is_null($name) ? $name : $options);

    110.

    111. 

  111.             $name = (!is_null($name) ? $name : url_for($options));
    112. 

  112.         }

    114. 

  114.         $html_options = array_merge($html_options, array("type" => "submit", "value" => $name));
    115. 

  115.         return "<form method=\"post\" action=\"" .  htmlspecialchars($url) . "\" class=\"button-to\"><div>" .
    116.

    124. 

  124.         $image_options = array("src" => (ereg("/", $src) ? $src : "/images/$src"));

  126.
