    32. 

  32.  *  Inflector contains static methods to convert English words between
    33. 

  33.  *  singular and plural, and phrases between the camel case form and
    34. 

  34.  *  the lower case underscore form.
    35. 

  35.  *
    36. 

  36.  *  @tutorial PHPonTrax/Inflector.cls

  38. class Inflector {

  40.     /**
    41. 

  41.      *  Rules for converting an English singular word to plural form
    42. 

  42.      */
    43. 

  43.     private static $plural_rules =
    44. 

  44.         array(  '/(x|ch|ss|sh)$/' => '\1es',            # search, switch, fix, box, process, address
    45. 

  45.                                 '/series$/' => '\1series',
    46. 

  46.                 '/([^aeiouy]|qu)ies$/' => '\1y',
    47. 

  47.                 '/([^aeiouy]|qu)y$/' => '\1ies',        # query, ability, agency
    48. 

  48.                                 '/(?:([^f])fe|([lr])f)$/' => '\1\2ves', # half, safe, wife
    49. 

  49.                                 '/sis$/' => 'ses',                      # basis, diagnosis
    50. 

  50.                                 '/([ti])um$/' => '\1a',                 # datum, medium
    51. 

  51.                                 '/person$/' => 'people',                # person, salesperson
    52. 

  52.                                 '/man$/' => 'men',                      # man, woman, spokesman
    53. 

  53.                                 '/child$/' => 'children',               # child
    54. 

  54.                                 '/(.*)status$/' => '\1statuses',
    55. 

  55.                 '/s$/' => 's',                          # no change (compatibility)
    56. 

  56.                                 '/$/' => 's'
    57. 

  57.         );
    58.

  59.     /**
    60. 

  60.      *  Rules for converting an English plural word to singular form
    61. 

  61.      */
    62. 

  62.     private static $singular_rules =
    63. 

  63.         array(  '/(x|ch|ss)es$/' => '\1',
    64. 

  64.                 '/movies$/' => 'movie',
    65. 

  65.                 '/series$/' => 'series',
    66. 

  66.                 '/([^aeiouy]|qu)ies$/' => '\1y',
    67. 

  67.                 '/([lr])ves$/' => '\1f',
    68. 

  68.                 '/([^f])ves$/' => '\1fe',
    69. 

  69.                 '/(analy|ba|diagno|parenthe|progno|synop|the)ses$/' => '\1sis',
    70. 

  70.                 '/([ti])a$/' => '\1um',
    71. 

  71.                 '/people$/' => 'person',
    72. 

  72.                 '/men$/' => 'man',
    73. 

  73.                 '/(.*)statuses$/' => '\1status',
    74. 

  74.                 '/children$/' => 'child',
    75. 

  75.                 '/news$/' => 'news',
    76. 

  76.                 '/s$/' => ''
    77. 

  77.         );

  79.     /**
    80. 

  80.      *  Pluralize a word according to English rules
    81. 

  81.      *
    82. 

  82.      *  Convert a lower-case singular word to plural form.
    83. 

  83.      *  @param  string $word  Word to be pluralized
    84. 

  84.      *  @return string  Plural of $word
    85. 

  85.      */
    86. 

  86.     function pluralize($word) {
    87. 

  87.         $original = $word;
    88. 

  88.         foreach(self::$plural_rules as $rule => $replacement) {
    89. 

  89.             $word = preg_replace($rule,$replacement,$word);
    90. 

  90.             if($original != $word) break;
    91. 

  91.         }
    92. 

  92.         return $word;
    93. 

  93.     }

    96. 

  96.      *  Singularize a word according to English rules

  97.      *

  98.      *  @param  string $word  Word to be singularized

  99.      *  @return string  Singular of $word

  101.     function singularize($word) {
    102. 

  102.         $original = $word;
    103. 

  103.         foreach(self::$singular_rules as $rule => $replacement) {
    104. 

  104.             $word = preg_replace($rule,$replacement,$word);
    105. 

  105.             if($original != $word) break;
    106. 

  106.         }
    107. 

  107.         return $word;
    108. 

  108.     }

    111. 

  111.      *  Capitalize a word making it all lower case with first letter uppercase

  112.      *
    113. 

  113.      *  @param  string $word  Word to be capitalized

  114.      *  @return string Capitalized $word

  116.     function capitalize($word) {

  117.         return ucfirst(strtolower($word));

  118.     }

  120.     /**
    121. 

  121.      *  Convert a phrase from the lower case and underscored form
    122. 

  122.      *  to the camel case form
    123. 

  123.      *
    124. 

  124.      *  @param string $lower_case_and_underscored_word  Phrase to
    125. 

  125.      *                                                   convert
    126. 

  126.      *  @return string  Camel case form of the phrase
    127. 

  127.      */
    128. 

  128.     function camelize($lower_case_and_underscored_word) {
    129. 

  129.         return str_replace(" ","",ucwords(str_replace("_"," ",$lower_case_and_underscored_word)));
    130. 

  130.     }

  132.     /**
    133. 

  133.      *  Convert a phrase from the camel case form to the lower case
    134. 

  134.      *  and underscored form
    135. 

  135.      *
    136. 

  136.      *  @param string $camel_cased_word  Phrase to convert
    137. 

  137.      *  @return string Lower case and underscored form of the phrase
    138. 

  138.      */
    139. 

  139.     function underscore($camel_cased_word) {
    140. 

  140.         $camel_cased_word = preg_replace('/([A-Z]+)([A-Z])/','\1_\2',$camel_cased_word);
    141. 

  141.         return strtolower(preg_replace('/([a-z])([A-Z])/','\1_\2',$camel_cased_word));
    142. 

  142.     }

  144.     /**
    145. 

  145.      *  Generate a more human version of a lower case underscored word
    146. 

  146.      *
    147. 

  147.      *  @param string $lower_case_and_underscored_word  A word or phrase in
    148. 

  148.      *                                            lower_case_underscore form
    149. 

  149.      *  @return string The input value with underscores replaced by
    150. 

  150.      *   blanks and the first letter of each word capitalized
    151. 

  151.      */
    152. 

  152.     function humanize($lower_case_and_underscored_word) {
    153. 

  153.         return ucwords(str_replace("_"," ",$lower_case_and_underscored_word));
    154. 

  154.     }

  156.     /**
    157. 

  157.      *  Convert a class name to the corresponding table name
    158. 

  158.      *
    159. 

  159.      *  The class name is a singular word or phrase in CamelCase.
    160. 

  160.      *  By convention it corresponds to a table whose name is a plural
    161. 

  161.      *  word or phrase in lower case underscore form.
    162. 

  162.      *  @param string $class_name  Name of {@link ActiveRecord} sub-class
    163. 

  163.      *  @return string Pluralized lower_case_underscore form of name
    164. 

  164.      */
    165. 

  165.     function tableize($class_name) {
    166. 

  166.         return self::pluralize(self::underscore($class_name));
    167. 

  167.     }

    169. 

  169.     /**
    170. 

  170.      *  Convert a table name to the corresponding class name
    171. 

  171.      *
    172. 

  172.      *  @param string $table_name Name of table in the database
    173. 

  173.      *  @return string Singular CamelCase form of $table_name
    174. 

  174.      */
    175. 

  175.     function classify($table_name) {
    176. 

  176.         return self::camelize(self::singularize($table_name));
    177. 

  177.     }

  179.     /**
    180. 

  180.      *  Get foreign key column corresponding to a table name

    183. 

  183.      *     key

  184.      *  @return string Column name of the foreign key column
    185. 

  185.      */
    186. 

  186.     function foreign_key($class_name) {
    187. 

  187.         return self::underscore($class_name) . "_id";
    188. 

  188.     }
