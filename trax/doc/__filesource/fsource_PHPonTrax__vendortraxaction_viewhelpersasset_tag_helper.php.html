  50.     private function compute_public_path($source, $dir, $ext) {
    51. 

  51.         if(!preg_match('/^[-a-z]+:\/\//', $source)) {
    52. 

  52.             if($source{0} != '/') {
    53. 

  53.                 $source = "/{$dir}/{$source}";
    54. 

  54.             }
    55. 

  55.             if(!strstr($source, '.')) {
    56. 

  56.                 $source = "{$source}.{$ext}";
    57. 

  57.             }
    58. 

  58.             if(!is_null(TRAX_URL_PREFIX)) {
    59. 

  59.                 $prefix = TRAX_URL_PREFIX;
    60. 

  60.                 if($prefix{0} != "/") {
    61. 

  61.                     $prefix = "/$prefix";
    62. 

  62.                 }
    63. 

  63.                 $source = $prefix . ((substr($prefix, -1) == "/") ? substr($source, 1) : $source);
    64. 

  64.             }            
    65. 

  65.         }
    66. 

  66.         return $this->controller_object->asset_host . $source;
    67. 

  67.     }

  94.     function javascript_include_tag() {
    95. 

  95.         if(func_num_args() > 0) {
    96. 

  96.             $sources = func_get_args();     
    97. 

  97.             $options = (is_array(end($sources)) ? array_pop($sources) : array());          
    98. 

  98.             if(in_array('defaults', $sources)) {
    99. 

  99.                 if(is_array($this->javascript_default_sources)) {
    100. 

  100.                     $sources = array_merge($this->javascript_default_sources, $sources);    
    101. 

  101.                 }                  
    102. 

  102.                 if(file_exists(TRAX_PUBLIC. "/javascripts/application.js")) {
    103. 

  103.                     $sources[] = 'application';
    104. 

  104.                 }
    105. 

  105.                 # remove defaults from array
    106. 

  106.                 unset($sources[array_search('defaults', $sources)]);  
    107. 

  107.             }
    108. 

  108.             $contents = array();
    109. 

  109.             foreach($sources as $source) {
    110. 

  110.                 $source = $this->javascript_path($source);
    111. 

  111.                 $contents[] = $this->content_tag("script", "", array_merge(array("type" => "text/javascript", "src" => $source), $options));
    112. 

  112.             }
    113. 

  113.             return implode("", $contents);
    114. 

  114.         }
    115. 

  115.     }

  166.     /**
    167. 

  167.      * Returns an image tag converting the $options instead html options on the tag, but with these special cases:
    168. 

  168.      *
    169. 

  169.      * * <tt>:alt</tt> - If no alt text is given, the file name part of the src is used (capitalized and without the extension)
    170. 

  170.      * * <tt>:size</tt> - Supplied as "XxY", so "30x45" becomes width="30" and height="45"
    171. 

  171.      *
    172. 

  172.      * The src can be supplied as a...
    173. 

  173.      * * full path, like "/my_images/image.gif"
    174. 

  174.      * * file name, like "rss.gif", that gets expanded to "/images/rss.gif"
    175. 

  175.      * * file name without extension, like "logo", that gets expanded to "/images/logo.png"
    176. 

  176.      */
    177. 

  177.     function image_tag($source, $options = array()) {
    178. 

  178.         $options['src'] = $this->image_path($source);
    179. 

  179.         $options['alt'] = $options['alt'] ? $options['alt'] : Inflector::capitalize(reset($file_array = explode('.', basename($options['src']))));
    180. 

  180.         
    181. 

  181.         if(isset($options['size'])) {
    182. 

  182.             $size = explode('x', $options["size"]);         
    183. 

  183.             $options['width'] = reset($size);
    184. 

  184.             $options['height'] = end($size);
    185. 

  185.             unset($options['size']);
    186. 

  186.         }
    187. 

  187.         
    188. 

  188.         return $this->tag("img", $options);
    189. 

  189.     }

  191.     /**
    192. 

  192.      * Returns a link tag that browsers and news readers can use to auto-detect a RSS or ATOM feed for this page. The $type can
    193. 

  193.      * either be <tt>:rss</tt> (default) or <tt>:atom</tt> and the $options follow the url_for() style of declaring a link target.
    194. 

  194.      *
    195. 

  195.      * Examples:
    196. 

  196.      *  auto_discovery_link_tag # =>
    197. 

  197.      *   <link rel="alternate" type="application/rss+xml" title="RSS" href="http://www.curenthost.com/controller/action" />
    198. 

  198.      *  auto_discovery_link_tag(:atom) # =>
    199. 

  199.      *   <link rel="alternate" type="application/atom+xml" title="ATOM" href="http://www.curenthost.com/controller/action" />
    200. 

  200.      *  auto_discovery_link_tag(:rss, {:action => "feed"}) # =>
    201. 

  201.      *   <link rel="alternate" type="application/rss+xml" title="RSS" href="http://www.curenthost.com/controller/feed" />
    202. 

  202.      *  auto_discovery_link_tag(:rss, {:action => "feed"}, {:title => "My RSS"}) # =>
    203. 

  203.      *   <link rel="alternate" type="application/rss+xml" title="My RSS" href="http://www.curenthost.com/controller/feed" />
    204. 

  204.      */
    205. 

  205.     function auto_discovery_link_tag($type = 'rss', $options = array(), $tag_options = array()) {
    206. 

  206.         return $this->tag(
    207. 

  207.           "link", array(
    208. 

  208.           "rel" => ($tag_options['rel'] ? $tag_options['rel'] : "alternate"),
    209. 

  209.           "type" => ($tag_options['type'] ? $tag_options['type'] : "application/{$type}+xml"),
    210. 

  210.           "title" => ($tag_options['title'] ? $tag_options['title'] : strtoupper($type)),
    211. 

  211.           "href" => url_for(array_merge($options, array('only_path' => false))))
    212. 

  212.         );
    213. 

  213.     }    
    214. 

  214. }
