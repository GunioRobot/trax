  402.

  404.         $browser_url = $_SERVER['REDIRECT_URL'];
    405. 

  405.         #if(strstr($_SERVER['REQUEST_URI'], "?"))
    406. 

  406.         #    $browser_url = substr($_SERVER['REQUEST_URI'], 0, strpos($_SERVER['REQUEST_URI'], "?"));
    407.

    408.

    409. 

  409.         
    410.

    411. 

  411.         # strip off url prefix, if any

    413. 

  413.             // FIXME: Do we know for sure that the
    414. 

  414.             // initial '/' will be there?

    420. 

  420.         // FIXME: Do we know for sure that the
    421. 

  421.         // initial '/' will be there?
    422.

    423. 

  423.         //error_log('browser url='.$browser_url);
    424. 

  424.  
    425. 

  425.         # strip trailing slash (if any)
    426. 

  426.         if(substr($browser_url, -1) == "/") {
    427. 

  427.             $browser_url = substr($browser_url, 0, -1);
    428. 

  428.         }
    429. 

  429.         //error_log('browser url='.$browser_url);
    430. 

  430.  
    431. 

  431.         if($browser_url) {
    432. 

  432.             $this->url_path = explode("/", $browser_url);
    433. 

  433.         } else {
    434. 

  434.             $this->url_path = array();
    435. 

  435.         }
    436. 

  436.  
    437. 

  437.         if($this->router->routes_count > 0) {
    438. 

  438.             $this->controllers_path = TRAX_ROOT . $GLOBALS['TRAX_INCLUDES']['controllers'];
    439. 

  439.             $this->helpers_path = $this->helpers_base_path = TRAX_ROOT . $GLOBALS['TRAX_INCLUDES']['helpers'];
    440. 

  440.             $this->application_controller_file = $this->controllers_path . "/application.php";
    441. 

  441.             $this->application_helper_file = $this->helpers_path . "/application_helper.php";
    442. 

  442.             $this->layouts_path = TRAX_ROOT . $GLOBALS['TRAX_INCLUDES']['layouts'];
    443. 

  443.             $this->views_path = TRAX_ROOT . $GLOBALS['TRAX_INCLUDES']['views'];
    444. 

  444.  
    445. 

  445.             $route = $this->router->find_route($browser_url);
    446. 

  446.  
    447. 

  447.             //  find_route() returns an array if it finds a path that
    448. 

  448.             //  matches the URL, null if no match found
    449. 

  449.             if(is_array($route)) {
    450.
