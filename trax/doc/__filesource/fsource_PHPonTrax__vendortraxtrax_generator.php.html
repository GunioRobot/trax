  204.     function run() {
    205. 

  205.  
    206. 

  206.         //  If command line arguments exist, parse them
    207. 

  207.         if (array_key_exists('argv', $_SERVER)) {
    208. 

  208.             if (array_key_exists(1, $_SERVER['argv'])) {
    209. 

  209.                 $command = strtolower($_SERVER["argv"][1]);
    210. 

  210.             }
    211. 

  211.             if (array_key_exists(2, $_SERVER['argv'])) {
    212. 

  212.                 $command_name = $_SERVER["argv"][2];
    213. 

  213.             }
    214. 

  214.         }
    215. 

  215.  
    216. 

  216.         //  Execute command or output a diagnostic
    217. 

  217.         if(empty($command)) {
    218. 

  218.             $this->generator_help();
    219. 

  219.         } else {
    220. 

  220.             switch($command) {
    221. 

  221.  
    222. 

  222.                 //  Process "controller" command
    223. 

  223.                 case "controller":
    224. 

  224.                     if(empty($command_name)) {
    225. 

  225.                         $this->controller_help();
    226. 

  226.                     } else {
    227. 

  227.                         $views = array();
    228. 

  228.                         if(array_key_exists(3, $_SERVER['argv'])
    229. 

  229.                            && ($_SERVER["argv"][3] != "")) {
    230. 

  230.                             for($i=3;$i < count($_SERVER["argv"]);$i++) {
    231. 

  231.                                $views[] = strtolower($_SERVER["argv"][$i]);
    232. 

  232.                             }
    233. 

  233.                         }
    234. 

  234.                         $this->generate_controller($command_name, $views);
    235. 

  235.                     }
    236. 

  236.                     break;
    237. 

  237.  
    238. 

  238.                 //  Process "model" command
    239. 

  239.                 //  $command_name is the name of the model
    240. 

  240.                 case "model":
    241. 

  241.                     if(empty($command_name)) {
    242. 

  242.                         $this->model_help();
    243. 

  243.                     } else {
    244. 

  244.                         $this->generate_model($command_name);
    245. 

  245.                     }
    246. 

  246.                     break;
    247. 

  247.  
    248. 

  248.                 //  Process "scaffold" command
    249. 

  249.                 //  $command_name has the name of the model
    250. 

  250.                 //  $_SERVER['argv'][3] has the name of the controller
    251. 

  251.                 case "scaffold":
    252. 

  252.  
    253. 

  253.                     //  Model name is required
    254. 

  254.                     if( empty($command_name) ) {
    255. 

  255.                         echo "Error: name of model omitted\n";
    256. 

  256.                         $this->scaffold_help();
    257. 

  257.                         break;
    258. 

  258.                     }
    259. 

  259.  
    260. 

  260.                     //  Controller name is optional
    261. 

  261.                     if (array_key_exists(3, $_SERVER["argv"])) {
    262. 

  262.                         $controller_name = $_SERVER["argv"][3];
    263. 

  263.                     } else {
    264. 

  264.                         $controller_name = null;
    265. 

  265.                     }
    266. 

  266.                         
    267. 

  267.                     //  Views are optional following controller name
    268. 

  268.                     $views = array();
    269. 

  269.                     if (array_key_exists(4, $_SERVER["argv"])
    270. 

  270.                         && ($_SERVER["argv"][4] != "")) {
    271. 

  271.                         for($i=4;$i < count($_SERVER["argv"]);$i++) {
    272. 

  272.                             $views[] = strtolower($_SERVER["argv"][$i]);
    273. 

  273.                         }
    274. 

  274.                     }
    275. 

  275.                     $this->generate_scaffold($command_name,
    276. 

  276.                                              $controller_name, $views);
    277. 

  277.                     break;                    
    278. 

  278.  
    279. 

  279.             default:
    280. 

  280.                 $this->generator_help();
    281. 

  281.             }                            // switch($command)
    282. 

  282.         }
    283. 

  283.         return;
    284. 

  284.     }

  295.      *  containing the class definition<br>
    296. 

  296.      *  <samp>class</samp> <i>SomeName</i><samp>Controller extends
    297. 

  297.      *  ApplicationController {}</samp></li>

    299. 

  299.      *  <samp>app/helpers/</samp><i>some_name</i><samp>_helper.php</samp></li>

  300.      *     <li>a directory
    301. 

  301.      *  <samp>app/views/</samp><i>some_name</i></li>
    302. 

  302.      *  </ul>
    303. 

  303.      *
    304. 

  304.      *  <p>Optionally, one or more views can be appended to the command:<br>
    305. 

  305.      *  <samp>php script/generate.php controller</samp>
    306. 

  306.      *  <i>SomeName view1 view2</i><br>
    307. 

  307.      *  which will additionally generate files:<br>
    308. 

  308.      *  <samp>app/views/</samp><i>some_name/view1</i><samp>.phtml</samp><br>

  309.      *  <samp>app/views/</samp><i>some_name/view2</i><samp>.phtml</samp></p>
    310. 

  310.      *
    311. 

  311.      *  @param string $name Name in CamelCase of the controller to generate.
    312. 

  312.      *                       The value may include '/' which will cause
    313. 

  313.      *                       creation of subdirectories indicated to
    314. 

  314.      *                       hold the controller and view files.
    315. 

  315.      *  @param string $views  Optional list of views to generate
    316. 

  316.      *  @param boolean $scaffolding 
    317. 

  317.      *  @uses Inflector::underscore()
    318. 

  318.      *  @uses $controller_class   Set during call
    319. 

  319.      *  @uses $controller_path    Must be set before call.
    320. 

  320.      *  @uses create_controller()
    321. 

  321.      *  @uses create_helper()
    322. 

  322.      *  @uses create_view()
    323. 

  323.      *  @uses $extra_path         Set during call
    324. 

  324.      *  @uses $helper_path        Must be set before call.
    325. 

  325.      *  @uses $layouts_path       Must be set before call.
    326. 

  326.      *  @uses $layout_filename    Set during call
    327. 

  327.      *  @uses $view_path          Must be set before call.
    328. 

  328.      */
    329. 

  329.     function generate_controller($name, $views="", $scaffolding = false) {
    330. 

  330.  
    331. 

  331.         # Set the View and Controller extra path info
    332. 

  332.         if(stristr($name, "/")) {
    333. 

  333.             $this->extra_path = substr($name,0,strrpos($name, "/"));
    334. 

  334.             $name = Inflector::underscore(substr($name,strrpos($name, "/")+1));
    335. 

  335.             $this->view_path .= "/$this->extra_path/$name";           
    336. 

  336.             $this->layouts_path .= "/$this->extra_path";
    337. 

  337.             $this->controller_path .= "/$this->extra_path";
    338. 

  338.             $this->helper_path .= "/$this->extra_path";
    339. 

  339.         } else {
    340. 

  340.             $name = Inflector::underscore($name);
    341. 

  341.             $this->view_path .= "/$name";
    342. 

  342.         }
    343. 

  343.         $this->layout_filename = $name;
    344. 

  344.  
    345. 

  345.         $this->controller_class = Inflector::camelize($name);
    346. 

  346.  
    347. 

  347.         # Create the extra folders for View / Controller
    348. 

  348.         if(!file_exists($this->view_path)) {
    349. 

  349.             $this->exec("$this->mkdir_cmd $this->view_path");
    350. 

  350.         }
    351. 

  351.  
    352. 

  352.         if(!file_exists($this->controller_path)) {
    353. 

  353.             $this->exec("$this->mkdir_cmd $this->controller_path");
    354. 

  354.         }
    355. 

  355.  
    356. 

  356.         if(!file_exists($this->helper_path)) {
    357. 

  357.             $this->exec("$this->mkdir_cmd $this->helper_path");
    358. 

  358.         }
    359. 

  359.  
    360. 

  360.         # Create the actual controller/helper files
    361. 

  361.         if(!$scaffolding) {
    362. 

  362.             $this->create_controller($name, $views);
    363. 

  363.         } 
    364. 

  364.         $this->create_helper($name);
    365. 

  365.  
    366. 

  366.         if($this->extra_path) {
    367. 

  367.             $name = $this->extra_path."/".$name;
    368. 

  368.         }
    369. 

  369.  
    370. 

  370.         # Create view files if any
    371. 

  371.         if(is_array($views)) {
    372. 

  372.             foreach($views as $view) {
    373. 

  373.                 $this->create_view($view,$name);
    374. 

  374.             }
    375. 

  375.         } elseif(!empty($views)) {
    376. 

  376.             $this->create_view($views,$name);
    377. 

  377.         }
    378. 

  378.     }

    389.

    390. 

  390.      *  @param string $name Name of the model.  May be in either
    391. 

  391.      *                 under_score or CamelCase.  If no '_' exists in
    392. 

  392.      *                 $name it is treated as CamelCase.
    393. 

  393.      *  @uses Inflector::underscore()
    394.

    399. 

  399.     function generate_model($name) {
    400. 

  400.  
    401. 

  401.         if(stristr($name, "_")) {
    402. 

  402.             $model_file = $this->model_path."/".strtolower($name).".php";
    403. 

  403.         } else {
    404. 

  404.             $model_file = $this->model_path."/".Inflector::underscore($name).".php";
    405. 

  405.         }
    406. 

  406.  
    407. 

  407.         $model_class = Inflector::camelize($name);
    408. 

  408.  
    409. 

  409.         if(!file_exists($model_file)) {
    410. 

  410.             if(file_exists($this->model_template_file)) {
    411. 

  411.                 $template = file_get_contents($this->model_template_file);
    412. 

  412.                 $template = str_replace('[class_name]',$model_class,$template);
    413. 

  413.                 if(!file_put_contents($model_file,$template)) {
    414. 

  414.                     echo "error creating model file: $model_file\n";
    415. 

  415.                 } else {
    416. 

  416.                     echo "created $model_file\n";
    417. 

  417.                     return true;
    418. 

  418.                 }
    419. 

  419.             } else {
    420. 

  420.                 echo "error model template file doesn't exist: $this->model_template_file\n";
    421. 

  421.             }
    422.

  651.
