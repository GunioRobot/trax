  94.     function run() {
    95. 

  95.         $command = strtolower($_SERVER["argv"][1]);
    96. 

  96.         $command_name = $_SERVER["argv"][2];
    97. 

  97.  
    98. 

  98.         if(empty($command)) {
    99. 

  99.             $this->generator_help();
    100. 

  100.         } else {
    101. 

  101.             switch($command) {
    102. 

  102.                 case "controller":
    103. 

  103.                     if(empty($command_name)) {
    104. 

  104.                         $this->controller_help();
    105. 

  105.                     } else {
    106. 

  106.                         if($_SERVER["argv"][3] != "") {
    107. 

  107.                             for($i=3;$i < count($_SERVER["argv"]);$i++) {
    108. 

  108.                                 $views[] = strtolower($_SERVER["argv"][$i]);
    109. 

  109.                             }
    110. 

  110.                         }
    111. 

  111.                         $this->generate_controller($command_name, $views);
    112. 

  112.                     }
    113. 

  113.                     break;
    114. 

  114.                 case "model":
    115. 

  115.                     if(empty($command_name)) {
    116. 

  116.                         $this->model_help();
    117. 

  117.                     } else {
    118. 

  118.                         $this->generate_model($command_name);
    119. 

  119.                     }
    120. 

  120.                     break;
    121. 

  121.                 case "scaffold":
    122. 

  122.                     if(empty($command_name)) {
    123. 

  123.                         $this->scaffold_help();
    124. 

  124.                     } else {
    125. 

  125.                         $controller_name = $_SERVER["argv"][3];
    126. 

  126.                         if($_SERVER["argv"][4] != "") {
    127. 

  127.                             for($i=4;$i < count($_SERVER["argv"]);$i++) {
    128. 

  128.                                 $views[] = strtolower($_SERVER["argv"][$i]);
    129. 

  129.                             }
    130. 

  130.                         }                        
    131. 

  131.                         $this->generate_scaffold($command_name, $controller_name, $views);
    132. 

  132.                     }
    133. 

  133.                     break;                    
    134. 

  134.             }
    135. 

  135.         }
    136. 

  136.         exit;
    137. 

  137.     }

  150.      *  ApplicationController {}</samp></li>

  151.      *     <li>a file
    152. 

  152.      *  <samp>app/helpers/</samp><i>some_name</i><samp>_helper.php</samp></li>

  153.      *     <li>a directory
    154. 

  154.      *  <samp>app/views/</samp><i>some_name</i></li>
    155. 

  155.      *  </ul>
    156. 

  156.      *
    157. 

  157.      *  <p>Optionally, one or more views can be appended to the command:<br>
    158. 

  158.      *  <samp>php script/generate.php controller</samp>
    159. 

  159.      *  <i>SomeName view1 view2</i><br>

  160.      *  which will additionally generate files:<br>
    161. 

  161.      *  <samp>app/views/</samp><i>some_name/view1</i><samp>.phtml</samp><br>
    162. 

  162.      *  <samp>app/views/</samp><i>some_name/view2</i><samp>.phtml</samp></p>
    163. 

  163.      *
    164. 

  164.      *  @param string $name Name of the controller to generate in camel case
    165. 

  165.      *  @param string $views  Optional list of views to generate
    166. 

  166.      *  @param boolean $scaffolding 
    167. 

  167.      */
    168. 

  168.     function generate_controller($name, $views="", $scaffolding = false) {
    169. 

  169.  
    170. 

  170.         # Set the View and Controller extra path info
    171. 

  171.         if(stristr($name, "/")) {
    172. 

  172.             $this->extra_path = substr($name,0,strrpos($name, "/"));
    173. 

  173.             $name = Inflector::underscore(substr($name,strrpos($name, "/")+1));
    174. 

  174.             $this->view_path .= "/$this->extra_path/$name";           
    175. 

  175.             $this->layouts_path .= "/$this->extra_path";
    176. 

  176.             $this->controller_path .= "/$this->extra_path";
    177. 

  177.             $this->helper_path .= "/$this->extra_path";
    178. 

  178.         } else {
    179. 

  179.             $name = Inflector::underscore($name);
    180. 

  180.             $this->view_path .= "/$name";
    181. 

  181.         }
    182. 

  182.         $this->layout_filename = $name;
    183. 

  183.  
    184. 

  184.         $this->controller_class = Inflector::camelize($name);
    185. 

  185.  
    186. 

  186.         # Create the extra folders for View / Controller
    187. 

  187.         if(!file_exists($this->view_path)) {
    188. 

  188.             $this->exec("$this->mkdir_cmd $this->view_path");
    189. 

  189.         }
    190. 

  190.  
    191. 

  191.         if(!file_exists($this->controller_path)) {
    192. 

  192.             $this->exec("$this->mkdir_cmd $this->controller_path");
    193. 

  193.         }
    194. 

  194.  
    195. 

  195.         if(!file_exists($this->helper_path)) {
    196. 

  196.             $this->exec("$this->mkdir_cmd $this->helper_path");
    197. 

  197.         }
    198. 

  198.  
    199. 

  199.         # Create the actual controller/helper files
    200. 

  200.         if(!$scaffolding) {
    201. 

  201.             $this->create_controller($name, $views);
    202. 

  202.         } 
    203. 

  203.         $this->create_helper($name);
    204. 

  204.  
    205. 

  205.         if($this->extra_path) {
    206. 

  206.             $name = $this->extra_path."/".$name;
    207. 

  207.         }
    208. 

  208.  
    209. 

  209.         # Create view files if any
    210. 

  210.         if(is_array($views)) {
    211. 

  211.             foreach($views as $view) {
    212. 

  212.                 $this->create_view($view,$name);
    213. 

  213.             }
    214. 

  214.         } elseif(!empty($views)) {
    215. 

  215.             $this->create_view($views,$name);
    216. 

  216.         }
    217. 

  217.     }

  231.     function generate_model($name) {
    232. 

  232.  
    233. 

  233.         if(stristr($name, "_")) {
    234. 

  234.             $model_file = $this->model_path."/".strtolower($name).".php";
    235. 

  235.         } else {
    236. 

  236.             $model_file = $this->model_path."/".Inflector::underscore($name).".php";
    237. 

  237.         }
    238. 

  238.  
    239. 

  239.         $model_class = Inflector::camelize($name);
    240. 

  240.  
    241. 

  241.         if(!file_exists($model_file)) {
    242. 

  242.             if(file_exists($this->model_template_file)) {
    243. 

  243.                 $template = file_get_contents($this->model_template_file);
    244. 

  244.                 $template = str_replace('[class_name]',$model_class,$template);
    245. 

  245.                 if(!file_put_contents($model_file,$template)) {
    246. 

  246.                     echo "error creating model file: $model_file\n";
    247. 

  247.                 } else {
    248. 

  248.                     echo "created $model_file\n";
    249. 

  249.                     return true;
    250. 

  250.                 }
    251. 

  251.             } else {
    252. 

  252.                 echo "error model template file doesn't exist: $this->model_template_file\n";
    253. 

  253.             }
    254. 

  254.         } else {
    255. 

  255.             echo "exists $model_file\n";
    256. 

  256.             return true;
    257. 

  257.         }
    258. 

  258.         return false;
    259. 

  259.     }
