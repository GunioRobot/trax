  386.                 break;

    392. 

  392.                 $parameters = is_array($this->has_and_belongs_to_many) ? $this->has_and_belongs_to_many[$key] : null;
    393. 

  393.                 $this->$key = $this->find_all_habtm($key, $parameters);

  413.     function __set($key, $value) {
    414. 

  414.         //echo "setting: $key = $value<br>";

  415.         if($key == "table_name") {
    416. 

  416.             $this->set_content_columns($value);
    417. 

  417.             # this elseif checks if first its an object if its parent is ActiveRecord
    418. 

  418.         } elseif(is_object($value) && get_parent_class($value) == __CLASS__ && $this->auto_save_associations) {
    419. 

  419.             if($association_type = $this->get_association_type($key)) {
    420. 

  420.                 $this->save_associations[$association_type][] = $value;
    421.
