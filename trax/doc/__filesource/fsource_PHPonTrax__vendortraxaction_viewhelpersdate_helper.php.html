  47.     private function check_request_for_value() {
    48. 

  48.         if(!$value = $_REQUEST[$this->object_name][$this->attribute_name]) {
    49. 

  49.             # check if this a date / datetime
    50. 

  50.             if($year_value = $_REQUEST[$this->object_name][$this->attribute_name."(1i)"]) {
    51. 

  51.                 $this->request_years[$this->attribute_name] = $year_value;    
    52. 

  52.             }
    53. 

  53.             if($month_value = $_REQUEST[$this->object_name][$this->attribute_name."(2i)"]) {
    54. 

  54.                 $this->request_months[$this->attribute_name] = $month_value;    
    55. 

  55.             }
    56. 

  56.             if($day_value = $_REQUEST[$this->object_name][$this->attribute_name."(3i)"]) {
    57. 

  57.                 $this->request_days[$this->attribute_name] = $day_value;    
    58. 

  58.             }
    59. 

  59.             if($minute_value = $_REQUEST[$this->object_name][$this->attribute_name."(4i)"]) {
    60. 

  60.                 $this->request_minutes[$this->attribute_name] = $minute_value;    
    61. 

  61.             }   
    62. 

  62.             if($hour_value = $_REQUEST[$this->object_name][$this->attribute_name."(5i)"]) {
    63. 

  63.                 $this->request_hours[$this->attribute_name] = $hour_value;    
    64. 

  64.             }              
    65. 

  65.             if($second_value = $_REQUEST[$this->object_name][$this->attribute_name."(6i)"]) {
    66. 

  66.                 $this->request_seconds[$this->attribute_name] = $second_value;    
    67. 

  67.             }                                                                   
    68. 

  68.         }   
    69. 

  69.         return $value;     
    70. 

  70.     }

  386.     function to_date_select_tag($options = array()) {

  388.         $options  = array_merge($defaults, $options);

  389.         $options_with_prefix = array();

    392. 

  392.         }        
    393. 

  393.         
    394. 

  394.         if($options['include_blank']) {

    398. 

  398.             $value = $this->value();

  399.             $date = $value ? $value : date("Y-m-d");

    401. 

  401.  
    402. 

  402.         $date_select = array();

  403.         if($options['month_before_year']) {
    404. 

  404.             $options['order'] = array('month', 'year', 'day');
    405. 

  405.         } elseif(!$options['order']) {
    406. 

  406.             $options['order'] = array('year', 'month', 'day');
    407. 

  407.         }
    408. 

  408.  
    409. 

  409.         $position = array('year' => 1, 'month' => 2, 'day' => 3);
    410. 

  410.  
    411. 

  411.         $discard = array();
    412. 

  412.         if($options['discard_year']) $discard['year']  = true;
    413. 

  413.         if($options['discard_month']) $discard['month'] = true;
    414. 

  414.         if($options['discard_day'] || $options['discard_month']) $discard['day'] = true;
    415. 

  415.  
    416. 

  416.         foreach($options['order'] as $param) {
    417. 

  417.             if(!$discard[$param]) {
    418. 

  418.                 $date_select[] = call_user_func(array($this, "select_$param"),  $date, $options_with_prefix[$position[$param]]);
    419. 

  419.             }
    420. 

  420.         }
    421. 

  421.         
    422. 

  422.         if(count($date_select)) {
    423. 

  423.             $seperator = $options['field_seperator'] ? $options['field_seperator'] : " ";
    424. 

  424.             $date_select = implode($seperator, $date_select);            
    425. 

  425.         }
    426. 

  426.  
    427.
