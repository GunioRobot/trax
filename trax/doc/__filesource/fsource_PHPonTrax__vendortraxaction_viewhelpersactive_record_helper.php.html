  106.     function form($record_name, $options = array()) {
    107. 

  107.         $record = $this->object($record_name);
    108. 

  108.         $options["action"] = $options[":action"] ? $options[":action"] : $record->is_new_record() ? "add" : "save";
    109. 

  109.         $action = url_for(array(':action' => $options[':action'], ':id' => $record->id));
    110. 

  110.         $submit_value = (isset($options['submit_value']) ? $options['submit_value'] : ucfirst(preg_replace('/[^\w]/', '', $options[':action'])));
    111. 

  111.  
    112. 

  112.         $contents = '';
    113. 

  113.         if(!$record->is_new_record()) $contents .= hidden_field($record_name, 'id');
    114. 

  114.         $contents .= $this->all_input_tags($record, $record_name, $options);
    115. 

  115.         if(isset($options['block'])) $contents .= eval($options['block']);
    116. 

  116.         $contents .= "<br>".submit_tag($submit_value)."<br><br>";
    117. 

  117.  
    118. 

  118.         return $this->content_tag('form', $contents, array('action' => $action, 'method' => 'post'));
    119. 

  119.     }

  160.     function error_messages_for($object_name, $options = array()) {
    161. 

  161.         if(is_object($object_name)) {
    162. 

  162.             $object_name = get_class($object_name);
    163. 

  163.             //echo "object name:".$object_name;

  164.         }
    165. 

  165.         $this->object_name = $object_name;
    166. 

  166.         $object = $this->controller_object->$object_name;
    167. 

  167.         if(!empty($object->errors)) {
    168. 

  168.             $errors = ($num_errors = count($object->errors)) > 1 ? "$num_errors errors" : "$num_errors error";
    169. 

  169.             return $this->content_tag("div",
    170. 

  170.                 $this->content_tag(
    171. 

  171.                     (isset($options['header_tag']) ? $options['header_tag'] : "h2"),
    172.

    174. 

  174.                 ) .
    175. 

  175.                 $this->content_tag("p", "There were problems with the following fields:") .
    176. 

  176.                 $this->content_tag("ul", array_reduce($object->errors, create_function('$v,$w', 'return ($v ? $v : "") . content_tag("li", $w);'), '')),
    177. 

  177.                 array("id" => (isset($options['id']) ? $options['id'] : "ErrorExplanation"), "class" => (isset($options['class']) ? $options['class'] : "ErrorExplanation"))

  233.     function to_tag($object_name, $attribute_name, $options = array()) {
    234. 

  234.         $this->object_name = $object_name;
    235. 

  235.         $this->attribute_name = $attribute_name;
    236. 

  236.         $form = new FormHelper($object_name, $attribute_name);
    237. 

  237.         switch($this->column_type()) {
    238. 

  238.         case 'string':
    239. 

  239.         case 'varchar':
    240. 

  240.         case 'varchar2':
    241. 

  241.             $field_type = (eregi("password", $this->attribute_name) ? "password" : "text");
    242. 

  242.             $results = $form->to_input_field_tag($field_type, $options);
    243. 

  243.             break;
    244. 

  244.  
    245. 

  245.         case 'text':
    246. 

  246.         case 'blob':
    247. 

  247.             $results = $form->to_text_area_tag($options);
    248. 

  248.             break;
    249. 

  249.  
    250. 

  250.         case 'integer':
    251. 

  251.         case 'int':
    252. 

  252.         case 'number':
    253. 

  253.         case 'float':
    254. 

  254.         case 'real':
    255. 

  255.             $results = $form->to_input_field_tag("text", $options);
    256. 

  256.             break;
    257. 

  257.  
    258. 

  258.         case 'date':
    259. 

  259.             $form = new DateHelper($object_name, $attribute_name);
    260. 

  260.             $results = $form->to_date_select_tag($options);
    261. 

  261.             break;
    262. 

  262.  
    263. 

  263.         case 'datetime':
    264. 

  264.         case 'timestamp':
    265. 

  265.             $results = $this->to_datetime_select_tag($options);
    266. 

  266.             break;
    267. 

  267.

    269. 

  269.         case 'bool':

  270.             $results = $form->to_boolean_select_tag($options);

  271.             break;
    272. 

  272.  
    273. 

  273.         }

  274.         if(count($this->object()->errors)) {
    275. 

  275.             $results = $this->error_wrapping($results, $this->object()->errors[$this->attribute_name]);
    276.

    277. 

  277.         return $results;      
    278. 

  278.     }
