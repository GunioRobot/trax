  266.     public function process($source) {
    267. 

  267.         // clean all elements in this array
    268. 

  268.         if (is_array($source)) {
    269. 

  269.             foreach($source as $key => $value) {
    270. 

  270.                 // for arrays in arrays
    271. 

  271.                 if (is_array($value)) $source[$key] = self::process($value);
    272. 

  272.                 // filter element for XSS and other 'bad' code etc.
    273. 

  273.                 if (is_string($value)) $source[$key] = self::remove(self::decode($value));
    274. 

  274.             }
    275. 

  275.             return $source;
    276. 

  276.         // clean this string
    277. 

  277.         } else if (is_string($source)) {
    278. 

  278.             // filter source for XSS and other 'bad' code etc.
    279. 

  279.             return self::remove(self::decode($source));
    280. 

  280.         // return parameter as given
    281. 

  281.         } else return $source;    
    282. 

  282.     }

    307. 

  307.      *  Inspect the input for tags "<tagname ...>" and check the tag
    308. 

  308.      *  name against a list of forbidden tag names.  Delete all tags
    309. 

  309.      *  with forbidden names.  If {@link $xssAuto} is true, delete all
    310. 

  310.      *  tags in {@link $tagBlacklist}.  If there is a user-defined tag
    311. 

  311.      *  list in {@link $tagsArray}, process according to the value of
    312. 

  312.      *  {@link $tagsMethod}.
    313. 

  313.      *
    314. 

  314.      *  If the tag name is OK, then call {@link filterAttr()} to check
    315. 

  315.      *  all attributes of the tag and delete forbidden attributes.
    316. 

  316.      *  @param string $source Input string to be 'cleaned'
    317. 

  317.      *  @return string Cleaned version of input parameter
    318. 

  318.      *  @uses filterAttr()
    319. 

  319.      *  @uses $tagBlacklist
    320. 

  320.      *  @uses $tagsArray
    321. 

  321.      *  @uses $tagsMethod
    322. 

  322.      *  @uses $xssAuto
    323. 

  323.      */
    324. 

  324.     protected function filterTags($source) {
    325. 

  325.         // filter pass setup
    326. 

  326.         $preTag = NULL;
    327. 

  327.         $postTag = $source;
    328. 

  328.         // find initial tag's position
    329. 

  329.         $tagOpen_start = strpos($source, '<');
    330. 

  330.         // interate through string until no tags left
    331. 

  331.         while($tagOpen_start !== FALSE) {
    332. 

  332.             // process tag interatively
    333. 

  333.             $preTag .= substr($postTag, 0, $tagOpen_start);
    334. 

  334.             $postTag = substr($postTag, $tagOpen_start);
    335. 

  335.             $fromTagOpen = substr($postTag, 1);
    336. 

  336.             // end of tag
    337. 

  337.             $tagOpen_end = strpos($fromTagOpen, '>');
    338. 

  338.             if ($tagOpen_end === false) break;
    339. 

  339.             // next start of tag (for nested tag assessment)
    340. 

  340.             $tagOpen_nested = strpos($fromTagOpen, '<');
    341. 

  341.             if (($tagOpen_nested !== false) && ($tagOpen_nested < $tagOpen_end)) {
    342. 

  342.                 $preTag .= substr($postTag, 0, ($tagOpen_nested+1));
    343. 

  343.                 $postTag = substr($postTag, ($tagOpen_nested+1));
    344. 

  344.                 $tagOpen_start = strpos($postTag, '<');
    345. 

  345.                 continue;
    346. 

  346.             } 
    347. 

  347.             $tagOpen_nested = (strpos($fromTagOpen, '<') + $tagOpen_start + 1);
    348. 

  348.             $currentTag = substr($fromTagOpen, 0, $tagOpen_end);
    349. 

  349.             $tagLength = strlen($currentTag);
    350. 

  350.             if (!$tagOpen_end) {
    351. 

  351.                 $preTag .= $postTag;
    352. 

  352.                 $tagOpen_start = strpos($postTag, '<');            
    353. 

  353.             }
    354. 

  354.             // iterate through tag finding attribute pairs - setup
    355. 

  355.             $tagLeft = $currentTag;
    356. 

  356.             $attrSet = array();
    357. 

  357.             $currentSpace = strpos($tagLeft, ' ');
    358. 

  358.             // is end tag
    359. 

  359.             if (substr($currentTag, 0, 1) == "/") {
    360. 

  360.                 $isCloseTag = TRUE;
    361. 

  361.                 list($tagName) = explode(' ', $currentTag);
    362. 

  362.                 $tagName = substr($tagName, 1);
    363. 

  363.             // is start tag
    364. 

  364.             } else {
    365. 

  365.                 $isCloseTag = FALSE;
    366. 

  366.                 list($tagName) = explode(' ', $currentTag);
    367. 

  367.             }        
    368. 

  368.             // excludes all "non-regular" tagnames OR no tagname OR remove if xssauto is on and tag is blacklisted
    369. 

  369.             if ((!preg_match("/^[a-z][a-z0-9]*$/i",$tagName)) || (!$tagName) || ((in_array(strtolower($tagName), self::$tagBlacklist)) && (self::$xssAuto))) {
    370. 

  370.                 $postTag = substr($postTag, ($tagLength + 2));
    371. 

  371.                 $tagOpen_start = strpos($postTag, '<');
    372. 

  372.                 // don't append this tag
    373. 

  373.                 continue;
    374. 

  374.             }
    375. 

  375.             // this while is needed to support attribute values with spaces in!
    376. 

  376.             while ($currentSpace !== FALSE) {
    377. 

  377.                 $fromSpace = substr($tagLeft, ($currentSpace+1));
    378. 

  378.                 $nextSpace = strpos($fromSpace, ' ');
    379. 

  379.                 $openQuotes = strpos($fromSpace, '"');
    380. 

  380.                 $closeQuotes = strpos(substr($fromSpace, ($openQuotes+1)), '"') + $openQuotes + 1;
    381. 

  381.                 // another equals exists
    382. 

  382.                 if (strpos($fromSpace, '=') !== FALSE) {
    383. 

  383.                     // opening and closing quotes exists
    384. 

  384.                     if (($openQuotes !== FALSE) && (strpos(substr($fromSpace, ($openQuotes+1)), '"') !== FALSE))
    385. 

  385.                         $attr = substr($fromSpace, 0, ($closeQuotes+1));
    386. 

  386.                     // one or neither exist
    387. 

  387.                     else $attr = substr($fromSpace, 0, $nextSpace);
    388. 

  388.                 // no more equals exist
    389. 

  389.                 } else $attr = substr($fromSpace, 0, $nextSpace);
    390. 

  390.                 // last attr pair
    391. 

  391.                 if (!$attr) $attr = $fromSpace;
    392. 

  392.                 // add to attribute pairs array
    393. 

  393.                 $attrSet[] = $attr;

  423.     /** 
    424. 

  424.      *  Internal method to strip a tag of certain attributes
    425. 

  425.      *
    426. 

  426.      *  Remove potentially dangerous attributes from a set of
    427. 

  427.      *  "attr=value" strings.  Attributes considered dangerous are:
    428. 

  428.      *  <ul>
    429. 

  429.      *    <li>Any attribute name containing any non-alphabetic
    430. 

  430.      *      character</li>
    431. 

  431.      *    <li>Any attribute name beginning "on..."</li>
    432. 

  432.      *    <li>If {@link $xssAuto} is true, any attribute name in
    433. 

  433.      *      {@link $attrBlacklist}</li>
    434. 

  434.      *    <li>Any attribute with a value containing the strings
    435. 

  435.      *      'javascript:', 'behaviour:', 'vbscript:', 'mocha:',
    436. 

  436.      *      'livescript:'</li>
    437. 

  437.      *    <li>Any attribute whose name contains 'style' and whose
    438. 

  438.      *      value contains 'expression'.</li>
    439. 

  439.      *    <li>If there is a user-provided list of attributes in
    440. 

  440.      *      {@link $attrArray}, process according to the value of
    441. 

  441.      *      {@link $attrMethod}.</li>
    442. 

  442.      *  </ul>
    443. 

  443.      *  @param string[] $attrSet Array of strings "attr=value" parsed
    444. 

  444.      *                            from a tag.
    445. 

  445.      *  @return string[] Input with potentially dangerous attributes
    446. 

  446.      *                    removed
    447. 

  447.      *  @uses $attrArray
    448. 

  448.      *  @uses $attrBlacklist
    449. 

  449.      *  @uses $attrMethod
    450. 

  450.      *  @uses $xssAuto
    451. 

  451.      */
    452. 

  452.     protected function filterAttr($attrSet) {    
    453. 

  453.         $newSet = array();
    454. 

  454.         // process attributes
    455. 

  455.         for ($i = 0; $i <count($attrSet); $i++) {
    456. 

  456.             // skip blank spaces in tag
    457. 

  457.             if (!$attrSet[$i]) continue;
    458. 

  458.             // split into attr name and value
    459. 

  459.             $attrSubSet = explode('=', trim($attrSet[$i]));
    460. 

  460.             list($attrSubSet[0]) = explode(' ', $attrSubSet[0]);
    461. 

  461.             // removes all "non-regular" attr names AND also attr blacklisted
    462. 

  462.             if ((!eregi("^[a-z]*$",$attrSubSet[0])) || ((self::$xssAuto) && ((in_array(strtolower($attrSubSet[0]), self::$attrBlacklist)) || (substr($attrSubSet[0], 0, 2) == 'on'))))
    463. 

  463.                 continue;
    464. 

  464.             // xss attr value filtering
    465. 

  465.             if ($attrSubSet[1]) {
    466. 

  466.                 // strips unicode, hex, etc
    467. 

  467.                 $attrSubSet[1] = str_replace('&#', '', $attrSubSet[1]);
    468. 

  468.                 // strip normal newline within attr value
    469. 

  469.                 $attrSubSet[1] = preg_replace('/\s+/', '', $attrSubSet[1]);
    470. 

  470.                 // strip double quotes
    471. 

  471.                 $attrSubSet[1] = str_replace('"', '', $attrSubSet[1]);
    472. 

  472.                 // [requested feature] convert single quotes from either side to doubles (Single quotes shouldn't be used to pad attr value)
    473. 

  473.                 if ((substr($attrSubSet[1], 0, 1) == "'") && (substr($attrSubSet[1], (strlen($attrSubSet[1]) - 1), 1) == "'"))
    474. 

  474.                     $attrSubSet[1] = substr($attrSubSet[1], 1, (strlen($attrSubSet[1]) - 2));
    475. 

  475.                 // strip slashes
    476. 

  476.                 $attrSubSet[1] = stripslashes($attrSubSet[1]);
    477. 

  477.             }
    478. 

  478.             // auto strip attr's with "javascript:
    479. 

  479.             if (    ((strpos(strtolower($attrSubSet[1]), 'expression') !== false) &&    (strtolower($attrSubSet[0]) == 'style')) ||
    480. 

  480.                     (strpos(strtolower($attrSubSet[1]), 'javascript:') !== false) ||
    481. 

  481.                     (strpos(strtolower($attrSubSet[1]), 'behaviour:') !== false) ||
    482. 

  482.                     (strpos(strtolower($attrSubSet[1]), 'vbscript:') !== false) ||
    483. 

  483.                     (strpos(strtolower($attrSubSet[1]), 'mocha:') !== false) ||
    484. 

  484.                     (strpos(strtolower($attrSubSet[1]), 'livescript:') !== false) 
    485. 

  485.             ) continue;
    486. 

  486.  
    487. 

  487.             // if matches user defined array
    488. 

  488.             $attrFound = in_array(strtolower($attrSubSet[0]), self::$attrArray);
    489. 

  489.             // keep this attr on condition
    490. 

  490.             if ((!$attrFound && self::$attrMethod) || ($attrFound && !self::$attrMethod)) {
    491. 

  491.                 // attr has value
    492. 

  492.                 if ($attrSubSet[1]) $newSet[] = $attrSubSet[0] . '="' . $attrSubSet[1] . '"';
    493. 

  493.                 // attr has decimal zero as value
    494. 

  494.                 else if ($attrSubSet[1] == "0") $newSet[] = $attrSubSet[0] . '="0"';
    495. 

  495.                 // reformat single attributes to XHTML
    496. 

  496.                 else $newSet[] = $attrSubSet[0] . '="' . $attrSubSet[0] . '"';
    497. 

  497.             }    
    498. 

  498.         }
    499. 

  499.         return $newSet;
    500. 

  500.     }

  514.     protected function decode($source) {
    515. 

  515.         // url decode
    516. 

  516.         $source = html_entity_decode($source, ENT_QUOTES, "ISO-8859-1");
    517.
